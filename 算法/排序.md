## 插入排序

```javascript
function insertSort(arr) {
    for (let i = 1; i < arr.length; i++) {
        for (let j = i; arr[j] < arr[j - 1]; j--) {
                let temp=arr[j]
                arr[j]=arr[j-1]
                arr[j-1]=temp
        }
    }
    return arr
}	
```

## 直接选择排序

```javascript
function selectSort(arr) {
    let mi,temp;
    for (let i = 0; i < arr.length; i++) {
        mi=i
        for (let j=i+1;j<arr.length;j++){
            if (arr[j]<arr[mi]){
                mi=j
            }
        }
        temp=arr[i]
        arr[i]=arr[mi]
        arr[mi]=temp
    }
    return arr
}
```
