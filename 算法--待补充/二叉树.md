## 二叉树遍历

```javascript
function TreeNode(val) {
	this.val=val;
    this.left=this.right=null
}
```

深度优先搜索是搜索算法的一种,它沿着树的深度遍历树的节点，尽可能深的搜索树的分支。

```javascript
//递归实现
function traverse(root){
    // 前序遍历
    traverse(root.Left)
    // 中序遍历
    traverse(root.Right)
    // 后序遍历
}

//前序非递归实现
function preOrder(root){
    if(!root){
        return []
    }
    let stack=[root]
    let ans=[]
    while(stack.length){
        let node=stack.pop()
        ans.push(node.val)
        node.right && stack.push(node.right)
        node.left && stack.push(node.left)
    }
    return ans
}
//中序非递归实现
function inOrder(root){
    if(!root){
        return []
    }
    let stack=[]
    let ans=[]
    while(stack.length || root){
        while(root){
            stack.push(root)
            root=root.left
        }
        root=stack.pop()
        ans.push(root.val)
        root=root.right
    }
    return ans
}
//后序非递归实现
function postOrder(root){
    if(!root){
        return []
    }
    let stack=[root]
    let ans=[]
    while(stack.length){
        let node=stack.pop()
        ans.unshift(node.val)
        node.left && stack.push(node.left)
        node.right && stack.push(node.right)
    }
    return ans
}
```

```javascript
function bfs(root){
    if(!root){
        return []
    }
    let ans=[]
    let stack=[root]
    while(stack.length){
        let len=stack.length
        for(let i=0;i<len;i++){
            let node=stack.shift()
            ans.push(node.val)
            node.left && stack.push(node.left)
            node.right && stack.push(node.right)
		}
    }
    return ans
}
```

```javascript
        let root=new TreeNode(rootVal)
```

## 中序和后序构建二叉树

```javascript
function buildTree(inorder,postorder){
    let m={}
    for(let i=0;i<inorder.length;i++){
        m[inorder[i]]=i
    }
    function help(pstart,pend,istart,iend){
        if(pstart>pend){
            return null
        }
        const rootVal=postorder(pend)
        const index=m[rootVal]
        const len=index-istart
        let root=new TreeNode(rootVal)
        root.left=help(pstart,pstart+len-1,istart,index)
        root.right=help(pstart+len,pend-1,index+1,iend)
        return root
    }
    return help(0, postorder.length - 1, 0, inorder.length - 1)
}
```

## 二叉树的深度

```javascript
//dfs
function maxDepth(root){
  const dfs=(root)=>{
  	if(!root){
        return 0
    }    
     let left=dfs(root.left)
     let right=dfs(root.right)
     return Math.max(left,right)+1
  }
  return dfs(root)
}

//bfs
function maxDepth(root){
    if(!root){
        return 0
    }
    let ans=0
    let stack=[root]
    while(stack.length){
        let len=stack.length
        while(len){
            let node=stack.shiift()
            node.left && stack.push(node.left)
            node.right && stack.push(node.right)
            len--
        }
        ans++
    }
    return ans
}
```
